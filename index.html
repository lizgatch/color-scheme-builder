Color scheme generator

You can use this editor to create your own color scheme for the jQuery snippet synthax highighter plugin. fork it on github.

JavaScript

$( document ).ready( runThis );

function runThis()
{
	// this script is for demoing purposes therefore and makes no sense
	
	var current_index = 2;
	var list = [ 'lorem', 'ipsum', 'dolor', 'sit', 'amet' ];
	var text = 'lorem ipsum dolor';
	
	for ( var i = 0; i < list.length; i++ )
	{
		if (
			i > current_index ||
			i > text.split( ' ' ).length
		)
		{
			text += ' ' + list[i];
		}
	}
	
	alert( text );
}
